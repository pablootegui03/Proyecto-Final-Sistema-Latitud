  <script>
    let procesando = false;

    function actualizarTitulo() {
      const select = document.getElementById('plantaSelect');
      const plantaTitle = document.getElementById('plantaTitle');
      
      if (select.value === 'JPV') {
        plantaTitle.textContent = 'JPV';
      } else if (select.value === 'RB') {
        plantaTitle.textContent = 'RB';
      } else {
        plantaTitle.textContent = 'Planta';
      }
    }

    function generarReporte() {
      if (procesando) return;

      const select = document.getElementById('plantaSelect');
      const planta = select.value;

      if (!planta) {
        mostrarMensaje('‚ö†Ô∏è Por favor, seleccione una planta', 'error');
        return;
      }

      procesando = true;
      deshabilitarUI(true);
      ocultarMensaje();
      mostrarProgreso(0, 'Iniciando proceso...');

      setTimeout(() => {
        actualizarProgreso(25, 'Ejecutando procesamiento en Azure...');
        
        google.script.run
          .withSuccessHandler(function(resultado) {
            manejarExito(resultado, planta);
          })
          .withFailureHandler(function(error) {
            manejarError(error);
          })
          .ejecutarFlujoReporte(planta);
      }, 300);
    }

    function manejarExito(resultado, planta) {
      if (resultado && resultado.error) {
        mostrarMensaje('‚ùå ' + resultado.error, 'error');
        resetearUI();
        return;
      }

      actualizarProgreso(50, 'Generando reporte en Azure...');

      setTimeout(() => {
        actualizarProgreso(75, 'Enviando reporte por email...');

        setTimeout(() => {
          actualizarProgreso(100, 'Proceso completado exitosamente');

          setTimeout(() => {
            mostrarMensaje(
              '‚úÖ Reporte de planta ' + planta + ' generado y enviado por email. Descargando...', 
              'success'
            );
            
            if (resultado && typeof resultado === 'string') {
              window.open(resultado, '_blank');
            }

            setTimeout(() => {
              resetearUI();
            }, 4000);
          }, 500);
        }, 800);
      }, 800);
    }

    function manejarError(error) {
      console.error('Error:', error);
      let mensaje = '‚ùå Error al generar el reporte';
      if (error && error.message) {
        mensaje += ': ' + error.message;
      }
      mostrarMensaje(mensaje, 'error');
      resetearUI();
    }

    function mostrarProgreso(porcentaje, mensaje) {
      const container = document.getElementById('progressContainer');
      container.style.display = 'block';
      actualizarProgreso(porcentaje, mensaje);
    }

    function actualizarProgreso(porcentaje, mensaje) {
      const bar = document.getElementById('progressBar');
      const messageEl = document.getElementById('progressMessage');
      bar.style.width = porcentaje + '%';
      messageEl.textContent = mensaje;
    }

    function ocultarProgreso() {
      const container = document.getElementById('progressContainer');
      container.style.display = 'none';
    }

    function mostrarMensaje(texto, tipo) {
      const mensaje = document.getElementById('mensajeResultado');
      mensaje.textContent = texto;
      mensaje.className = 'mensaje-resultado mensaje-' + tipo;
      mensaje.style.display = 'block';
    }

    function ocultarMensaje() {
      const mensaje = document.getElementById('mensajeResultado');
      mensaje.style.display = 'none';
    }

    function deshabilitarUI(deshabilitar) {
      const select = document.getElementById('plantaSelect');
      const btn = document.getElementById('btnGenerar');
      
      select.disabled = deshabilitar;
      btn.disabled = deshabilitar;
      
      if (deshabilitar) {
        btn.textContent = 'PROCESANDO...';
        btn.classList.add('btn-disabled');
      } else {
        btn.textContent = 'GENERAR REPORTE';
        btn.classList.remove('btn-disabled');
      }
    }

    function resetearUI() {
      procesando = false;
      deshabilitarUI(false);
      setTimeout(() => {
        ocultarProgreso();
      }, 500);
    }
  </script>
